# C. Challenge Payment Question
The Foodie-Fi team wants you to create a new payments table for the year 2020 that includes amounts paid by each customer in the subscriptions table with the following requirements:

    - monthly payments always occur on the same day of month as the original start_date of any monthly paid plan
    - upgrades from basic to monthly or pro plans are reduced by the current paid amount in that month and start immediately
    - upgrades from pro monthly to pro annual are paid at the end of the current billing period and also starts at the end of the month period
    - once a customer churns they will no longer make payments

### Create payments table.
```sql
drop table if exists payments;

create table payments(
  customer_id INTEGER,
  plan_id INTEGER,
  plan_name VARCHAR(20),
  payment_date DATE,
  amount NUMERIC,
  payment_order INTEGER
);

insert into payments(customer_id, plan_id, plan_name, payment_date, amount, payment_order)
with recursive leaded as (
  SELECT
	s.customer_id,
    s.plan_id,
    p.plan_name,
    s.start_date as payment_date,
    lead(s.start_date) over(partition by s.customer_id order by s.start_date) as next_date,
    p.price as current_amount
from
	food.subscriptions s
join
	food.plans p on
    s.plan_id = p.plan_id
where
	extract(year from s.start_date) = 2020 and
    p.plan_name != 'trial'
),
monthly_payments as(
	select
  		customer_id,
  		plan_id,
  		plan_name,
  		payment_date,
		current_amount,
		next_date
  	from
  		leaded
 	union
  	select
  		customer_id,
  		plan_id,
  		plan_name,
		(payment_date + interval '1 month')::date,
		current_amount,
		next_date
  	from
  		monthly_payments
  	where
		(payment_date + interval '1 month')::date < next_date
),
basic_to_pro as (
	select
		customer_id,
		plan_id,
		lag(plan_id) over(partition by customer_id order by payment_date) as previous_plan_id,
		plan_name,
		lag(plan_name) over(partition by customer_id order by payment_date) as previous_plan_name,
		current_amount,
		lag(current_amount) over(partition by customer_id order by payment_date) as previous_amount,
		payment_date,
		lag(payment_date) over(partition by customer_id order by payment_date) as previous_payment_date
	from
		monthly_payments
)
select
	customer_id,
	plan_id,
	plan_name,
	payment_date,
	case 
		when previous_plan_id < plan_id and previous_plan_id = 1 then current_amount - previous_amount
		else current_amount
	end as amount,
	row_number() over(partition by customer_id order by payment_date) as payment_order
from
	basic_to_pro
where
	plan_name != 'churn';
```
